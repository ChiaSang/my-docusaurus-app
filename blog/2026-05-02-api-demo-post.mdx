import {useEffect, useState} from 'react';

export function ApiResult() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState('');

  useEffect(() => {
    const controller = new AbortController();

    fetch('https://free.mockerapi.com/get', {signal: controller.signal})
      .then((response) => {
        if (!response.ok) {
          throw new Error(`请求失败：${response.status}`);
        }
        return response.json();
      })
      .then(setData)
      .catch((err) => {
        if (err?.name !== 'AbortError') {
          setError(err?.message || '请求失败');
        }
      })
      .finally(() => setLoading(false));

    return () => controller.abort();
  }, []);

  if (loading) {
    return <p>正在加载接口数据…</p>;
  }

  if (error) {
    return <p style={{color: 'var(--ifm-color-danger)'}}>加载失败：{error}</p>;
  }

  return <pre>{JSON.stringify(data, null, 2)}</pre>;
}

# 用 MDX 写一个交互式 API 博客

这篇文章演示如何在博客里直接写 React 组件，并请求 `https://free.mockerapi.com/get`，把返回内容渲染出来。
